### Add a page

<CH.Section>

Pages are just NextJS pages with two files

1. [MDX content.mdx](focus://content.mdx#1:20) with an [exported meta](focus://content.mdx#1:4) and [MDX afterwards](focus://content.mdx#6:10).
1. [NextJS page.tsx](focus://page.tsx#1:20)

<CH.Code>

```mdx content.mdx
export const meta = {
  title: "Page Title",
  weight: 1
};

# { meta.title }

page content
```

```tsx page.tsx
"use client";

// mark
import Content from "./content.mdx";

export default function Page(props: any) {
  return (
    // mark
    <Content {...props} />
  );
}
```


</CH.Code>

</CH.Section>


You can size and style the content by wrapping it in a `<div>`.

<CH.Code>

```tsx wrapped.tsx mark=7:9
"use client";

import Content from "./content.mdx";

export default function Page(props: any) {
  return (
    <div className="p-8">
      <Content {...props} />
    </div>
  );
}
```

</CH.Code>

It's often better to do this in a layout for consistent styling across the pages

```tsx layout.tsx
// from ./layout.tsx
```
